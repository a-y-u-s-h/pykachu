```
│   ├── core

│   ├── dependencies
│   └── logs
├── docs
│   ├── api
│   └── contribution
├── src
│    ├── automation
│    │   ├── builder
│    │   ├── cleaner
│    │   ├── deployer
│    │   └── docker
│    ├── core
│    │   ├── classes
│    │   ├── engines

│    │   └── scripts
│    ├── dependencies

```
